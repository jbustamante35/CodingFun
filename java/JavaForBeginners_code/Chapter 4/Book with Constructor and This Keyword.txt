class Book {
	final static int DEFAULT_YEAR = 2014;
	final String title;
	final int releaseYear;
	int copiesSold;
	
	Book(String t) {
		this(t, DEFAULT_YEAR, 0);
	}
	
	Book(String t, int r) {
		this(t, r, 0);
	}
	
	Book(String t, int r, int s) {
		title = t;
		releaseYear = r;
		copiesSold = s;
	}
}
