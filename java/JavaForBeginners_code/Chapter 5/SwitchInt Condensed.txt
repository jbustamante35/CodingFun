public class SwitchInt {
	public static void main(String[] args) {
		int month = 4; // here 4 represents April
		int lastDay;
		boolean leapYear = false;

				switch (month) {
		case 1:
		case 3:
		case 5:
		case 7:
		case 8:
		case 10:
		case 12:
			lastDay = 31;
			break;
		case 2:
			if (leapYear == true) {
				lastDay = 29;
			} else {
				lastDay = 28;
			}
			break;
		case 4:
		case 6:
		case 9:
		case 11:
			lastDay = 30;
			break;
		default:
			lastDay = 0;
		}
	}
}
